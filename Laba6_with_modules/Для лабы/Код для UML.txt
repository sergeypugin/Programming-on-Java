' Kлассы сгруппированы по модулям для наглядности

package "common.data" {
  class Product implements Comparable<Product> {
    - Integer id
    - String name
    - Coordinates coordinates
    - Date creationDate
    - long price
    - UnitOfMeasure unitOfMeasure
    - Person owner
    + void setId(Integer)
    + Integer getId()
    + String getName()
    + Coordinates getCoordinates()
    + Date getCreationDate()
    + long getPrice()
    + UnitOfMeasure getUnitOfMeasure()
    + Person getOwner()
  }

  class Coordinates {
    - double x
    - Float y
    + double getX()
    + Float getY()
  }

  class Person {
    - String name
    - Integer height
    - float weight
    + String getName()
    + Integer getHeight()
    + float getWeight()
  }

  enum UnitOfMeasure {
    SQUARE_METERS
    MILLILITERS
    GRAMS
  }

  Product o-- Coordinates
  Product o-- Person
  Product o-- UnitOfMeasure
}

package "common.forCommunicate" {
  class Request {
    - String commandName
    - String argument
    - Product objectArgument
    + String getCommandName()
    + String getArgument()
    + Product getObjectArgument()
  }

  class Response {
    - String message
    - boolean success
    + String getMessage()
    + boolean isSuccess()
  }

  class SerializationUtils {
    + {static} byte[] serialize(Object)
    + {static} Object deserialize(byte[])
  }

  Request o-- Product
}

package "client" {
  class ClientMain {
    - {static} int serverPort
    - {static} String serverHost
    - {static} Console console
    - {static} DatagramChannel channel
    + {static} void main(String[])
    - {static} void processCommand(String)
    - {static} void executeScript(String)
    - {static} void sendAndReceive(Request)
  }

  class Console {
    - Scanner scanner
    + Product askProduct()
    - String askName()
    - Coordinates askCoordinates()
    - long askPrice()
    - UnitOfMeasure askUnitOfMeasure()
    - Person askOwner()
  }

  ClientMain ..> Console : uses
  ClientMain ..> Request : creates
  ClientMain ..> Response : uses
  ClientMain ..> SerializationUtils : uses
  Console ..> Product : creates
  Console ..> Coordinates : creates
  Console ..> Person : creates
}

package "server" {
  class ServerMain {
    - {static} int port
    + {static} void main(String[])
    - {static} String findLatestCollectionFile()
  }

  class CollectionManager {
    - LinkedList<Product> collection
    - Date creationDate
    - int lastId
    + LinkedList<Product> getCollection()
    + void add(Product)
    + void clear()
    + boolean removeById(int)
    + boolean replace(int, Product)
  }

  class CommandManager {
    - Map<String, Command> map
    + Response reply(Request)
  }

  class FileManager {
    + {static} void writeCollectionToXML(String, LinkedList<Product>)
    + {static} LinkedList<Product> readCollectionFromXML(String)
  }

  ServerMain ..> CommandManager : creates and uses
  ServerMain ..> CollectionManager : creates and uses
  ServerMain ..> FileManager : uses
  ServerMain ..> Request : uses
  ServerMain ..> Response : creates
  ServerMain ..> SerializationUtils : uses
  CommandManager o-- Command
  CollectionManager o-- Product
}

package "server.commands" {
  interface Command {
    + Response execute(Request)
  }

  class Add implements Command {
    - CollectionManager collectionManager
  }
  class Clear implements Command {
    - CollectionManager collectionManager
  }
  class Count_by_unit_of_measure implements Command {
    - CollectionManager cm
  }
  class Filter_by_owner implements Command {
    - CollectionManager cm
  }
  class Help implements Command {}
  class Info implements Command {
    - CollectionManager cm
  }
  class Print_field_descending_unit_of_measure implements Command {
    - CollectionManager cm
  }
  class Remove_at implements Command {
    - CollectionManager collectionManager
  }
  class Remove_by_id implements Command {
    - CollectionManager collectionManager
  }
  class Remove_last implements Command {
    - CollectionManager cm
  }
  class Reorder implements Command {
    - CollectionManager collectionManager
  }
  class Save {
   - CollectionManager cm
   + void execute()
  }
  class Show implements Command {
    - CollectionManager cm
  }
  class Update implements Command {
    - CollectionManager cm
  }

  Command <|.. Add
  Command <|.. Clear
  Command <|.. Count_by_unit_of_measure
  Command <|.. Filter_by_owner
  Command <|.. Help
  Command <|.. Info
  Command <|.. Print_field_descending_unit_of_measure
  Command <|.. Remove_at
  Command <|.. Remove_by_id
  Command <|.. Remove_last
  Command <|.. Reorder
  Command <|.. Show
  Command <|.. Update

  Add ..> CollectionManager
  Clear ..> CollectionManager
  Count_by_unit_of_measure ..> CollectionManager
  Filter_by_owner ..> CollectionManager
  Info ..> CollectionManager
  Print_field_descending_unit_of_measure ..> CollectionManager
  Remove_at ..> CollectionManager
  Remove_by_id ..> CollectionManager
  Remove_last ..> CollectionManager
  Reorder ..> CollectionManager
  Save ..> CollectionManager
  Save ..> FileManager
  Show ..> CollectionManager
  Update ..> CollectionManager
}
